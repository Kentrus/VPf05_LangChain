def _lc_to_openai_messages(messages: List[BaseMessage]) -> List[Dict[str, str]]:
    role_map = {
        SystemMessage: "system",
        HumanMessage: "user",
        AIMessage: "assistant",
    }
    converted: List[Dict[str, str]] = []
    for m in messages:
        role = None
        for cls, r in role_map.items():
            if isinstance(m, cls):
                role = r
                break
        if role is None:
            # Fallback to user
            role = "user"
        content = m.content if isinstance(m.content, str) else str(m.content)
        converted.append({"role": role, "content": content})
    return converted

def get_llm() -> RunnableLambda:
    """Create a Runnable that calls OpenAI Chat Completions API directly.

    Environment:
    - OPENAI_API_KEY: API key (required)
    - OPENAI_BASE_URL: Optional custom base URL
    - OPENAI_MODEL or MODEL: Model name (default: gpt-4o-mini)
    - Optional: TEMPERATURE (float, default 0.2)
    """
    api_key = os.getenv("OPENAI_API_KEY")
    base_url = os.getenv("OPENAI_BASE_URL")
    model = os.getenv("OPENAI_MODEL") or os.getenv("MODEL") or "gpt-4o-mini"
    try:
        temperature = float(os.getenv("TEMPERATURE", "0.2"))
    except ValueError:
        temperature = 0.2

    if not api_key:
        raise RuntimeError("OPENAI_API_KEY is not set. Please set it to use the OpenAI client.")

def run_chain(task: str, out_dir: Path) -> Path:
    llm = get_llm()

    analysis_chain = build_analysis_chain(llm)
    code_chain = build_code_chain(llm)
    review_chain = build_review_chain(llm)

    # 1) Analysis
    analysis_text = analysis_chain.invoke({"task": task})
    try:
        spec = parse_json_spec(analysis_text)
    except Exception:
        # Fallback: enforce JSON via a stricter re-ask

            ("system", "Return ONLY compact JSON for the bot spec. No prose."),
            ("human", "Bot description: {task}"),
        ])

    # 2) Code generation
    code_text = code_chain.invoke({"spec_json": spec_json})
    code_text = strip_code_fences(code_text).strip()

    # 3) Minimal review and one-pass correction if needed
    issues = minimal_static_review(code_text)
    if issues:
        fixed = review_chain.invoke({"issues": "\\n".join(issues), "code": code_text})

        if len(fixed) > 0 and (len(fixed_issues) < len(issues)):
            code_text = fixed

    # Final review before writing
